      def initialize(options={})
        raise ArgumentError, 'No return url provided.'  unless options[:return]
        warn  'No trust url provided.'  unless options[:trust]
        options[:trust] ||= options[:return]

        @options  = {
          :session_key => 'rack.session'
        }.merge(options)
      end
