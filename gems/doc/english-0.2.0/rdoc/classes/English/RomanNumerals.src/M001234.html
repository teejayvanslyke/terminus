  def from_integer(integer)
    return nil if integer < 0 || integer > MAX

    r = integer  # remainder
    y = ''       # result

    NUMBERS.each do |n|
      while r >= n
        r -= n
        y += LOOKUP[n]
      end
      break if r <= 0
    end

    return y
  end
