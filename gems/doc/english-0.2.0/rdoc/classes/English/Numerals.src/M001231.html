  def name(integer)
    places = integer.to_i.to_s.split(//).collect{|s| s.to_i}.reverse
    name = []
    ((places.length + 2) / 3).times do |p|
      strings = Numerals.trio(places[p * 3, 3])
      name.push(Mega[p]) if strings.length > 0 and p > 0
      name += strings
    end
    name.push(Ones[0]) unless name.length > 0
    name.reverse.join(" ")
  end
