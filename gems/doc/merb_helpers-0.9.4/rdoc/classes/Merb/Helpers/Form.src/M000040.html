      def radio_control(col, attrs = {})
        errorify_field(attrs, col)
        attrs.merge!(:id => control_id(col))
        val = @_obj.send(col)
        attrs.merge!(:checked => "checked") if val.to_s == attrs[:value]
        optional_label(attrs) { radio_field(control_name_value(col, attrs)) }
      end
