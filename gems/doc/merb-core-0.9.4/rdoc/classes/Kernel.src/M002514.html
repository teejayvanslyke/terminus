  def load_dependencies(*args)
    args.each do |arg|
      case arg
      when String then load_dependency(arg)
      when Hash   then arg.each { |r,v| load_dependency(r, v) }
      when Array  then arg.each { |r|   load_dependency(r)    }
      end
    end
  end
