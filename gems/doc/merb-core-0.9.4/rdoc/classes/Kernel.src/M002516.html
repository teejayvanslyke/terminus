  def use_orm(orm)
    begin
      register_orm(orm)
      orm_plugin = "merb_#{orm}"
      Kernel.dependency(orm_plugin)
    rescue LoadError => e
      Merb.logger.warn!("The #{orm_plugin} gem was not found.  You may need to install it.")
      raise e
    end
  end
