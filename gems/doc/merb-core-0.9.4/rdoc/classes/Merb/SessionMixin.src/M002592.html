    def set_session_id_cookie(key)
      options = {}
      options[:value] = key
      options[:expires] = Time.now + _session_expiry if _session_expiry > 0
      options[:domain] = _session_cookie_domain if _session_cookie_domain
      cookies[_session_id_key] = options
    end
