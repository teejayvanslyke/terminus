      def merged_placeholders
        placeholders = {}
        (ancestors.reverse + [self]).each do |a|
          a.placeholders.each_pair do |k, pair|
            param, place = pair
            placeholders[k] = [param, place + (param == :path ? a.total_previous_captures : 0)]
          end
        end
        placeholders
      end
