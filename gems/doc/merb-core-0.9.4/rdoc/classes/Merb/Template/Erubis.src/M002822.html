    def self.compile_template(io, name, mod)
      template = ::Erubis::BlockAwareEruby.new(io.read)

      _old_verbose, $VERBOSE = $VERBOSE, nil
      template.def_method(mod, name, File.expand_path(io.path))
      $VERBOSE = _old_verbose
      
      name
    end
