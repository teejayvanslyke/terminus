    def regenerate
      new_sid = Merb::SessionMixin::rand_uuid 
      old_sid = @session_id
      MemorySessionContainer[new_sid] = MemorySessionContainer[old_sid]
      @session_id = new_sid
      MemorySessionContainer.delete(old_sid)
      self.needs_new_cookie=true 
    end
