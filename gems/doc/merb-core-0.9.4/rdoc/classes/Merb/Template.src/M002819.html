    def inline_template(io, mod = Merb::InlineTemplates)
      path = File.expand_path(io.path)
      ret = METHOD_LIST[path.gsub(/\.[^\.]*$/, "")] = 
        engine_for(path).compile_template(io, template_name(path), mod)
      io.close
      ret
    end
