    def matches?(stringlike, &blk)
      @document = case stringlike
      when Hpricot::Elem
        stringlike
      when StringIO
        Hpricot.parse(stringlike.string)
      else
        Hpricot.parse(stringlike)
      end
      
      @blk = blk unless blk.nil?

      unless @blk.nil?
        !@document.search(selector).select do |ele|
          @blk.call ele
          true
        end.empty?
      else
        !@document.search(selector).empty?
      end
    end
