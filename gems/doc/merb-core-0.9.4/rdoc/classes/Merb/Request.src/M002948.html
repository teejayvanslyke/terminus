      def query_parse(qs, d = '&;', preserve_order = false)
        qh = preserve_order ? Dictionary.new : {}
        (qs||'').split(/[#{d}] */n).inject(qh) { |h,p| 
          key, value = unescape(p).split('=',2)
          normalize_params(h, key, value)
        }
        preserve_order ? qh : qh.to_mash
      end
