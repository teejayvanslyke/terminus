      def self.included(base)
        base.send :include, ActiveSupport::Callbacks
        base.define_callbacks :setup, :teardown

        begin
          require 'mocha'
          base.alias_method_chain :run, :callbacks_and_mocha
        rescue LoadError
          base.alias_method_chain :run, :callbacks
        end
      end
