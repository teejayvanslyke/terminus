    def fetch_hash(with_table=nil)
      row = fetch_row
      return if row == nil
      hash = {}
      @fields.each_index do |i|
        f = with_table ? @fields[i].table+"."+@fields[i].name : @fields[i].name
        hash[f] = row[i]
      end
      hash
    end
