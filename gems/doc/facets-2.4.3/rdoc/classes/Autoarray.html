AutoArray

An Array that automatically expands dimensions as needed. 

  a  = Autoarray.new
  a[1][2][3] = 12
  a             #=> [nil, [nil, nil, [nil, nil, nil, 12]]]
  a[2][3][4]    #=> []
  a             #=> [nil, [nil, nil, [nil, nil, nil, 12]]]
  a[1][-2][1] = "Negative"
  a             #=> [nil, [nil, [nil, "Negative"], [nil, nil, nil, 12]]]

  def [](k)
    if -self.length+1 < k and k < self.length
      super(k)
    else
      Autoarray.new(0, nil, self, k)
    end
  end
